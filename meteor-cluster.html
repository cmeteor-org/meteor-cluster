Welcome to Meteor Cluster!

In brower console you can execute `some code` at server. Example:

Meteor.call('exec', "(function(){return process.env.ROOT_URL + '|' + process.pid;})()",function(err, rel) {console.log('root url:', rel)});

Meteor.call('exec', "( function(){return require('os').cpus().length;} )()", function(err, rel) { console.log('os cpus nums:', rel); });
